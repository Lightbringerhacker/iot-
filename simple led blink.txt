import RPi.GPIO as GPIO
import time

GPIO.setmode(GPIO.BOARD)
GPIO.setup(35, GPIO.OUT)
GPIO.setup(12, GPIO.OUT)

while True:
    GPIO.output(35, True)
    time.sleep(1)
    
    GPIO.output(35, False)
    time.sleep(1)
    
    GPIO.output(12, True)
    time.sleep(1)
    
    GPIO.output(12, False)
    time.sleep(1)
